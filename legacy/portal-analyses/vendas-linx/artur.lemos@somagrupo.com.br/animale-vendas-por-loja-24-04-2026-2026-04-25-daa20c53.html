Venda do dia (físico, RL_DESTINO=1)
SELECT
  v.CODIGO_FILIAL_DESTINO, ANY_VALUE(v.FILIAL_DESTINO) AS filial_nome,
  SUM(SAFE_CAST(v.VALOR_PAGO_PROD AS NUMERIC)) AS venda_liquida,
  SUM(v.QTDE_PROD) AS pecas,
  COUNT(DISTINCT COALESCE(NULLIF(v.PEDIDO_SITE,''),
    CONCAT(FORMAT_DATE('%Y%m%d', v.DATA_VENDA), '|',
           CAST(v.CODIGO_FILIAL_ORIGEM AS STRING), '|', v.TICKET))) AS atendimentos
FROM `soma-pipeline-prd.silver_linx.TB_WANMTP_VENDAS_LOJA_CAPTADO` v
WHERE v.DATA_VENDA = DATE '2026-04-24'
  AND CAST(v.RL_DESTINO AS STRING) = '1'
  AND v.TIPO_VENDA = 'VENDA_LOJA'
  AND CAST(v.CODIGO_FILIAL_DESTINO AS STRING) != '190454'
GROUP BY 1 ORDER BY venda_liquida DESC
Cota do dia (Animale)
SELECT pv.FILIAL, f.COD_FILIAL,
  SUM(SAFE_CAST(pv.PREVISAO_VALOR AS NUMERIC)) AS meta_dia
FROM `soma-pipeline-prd.silver_linx.LOJAS_PREVISAO_VENDAS` pv
LEFT JOIN `soma-pipeline-prd.silver_linx.FILIAIS` f ON pv.FILIAL = f.FILIAL
WHERE pv.DATA_VENDA = DATE '2026-04-24'
  AND f.REDE_LOJAS = '1'
  AND CAST(f.COD_FILIAL AS STRING) != '190454'
GROUP BY 1, 2 ORDER BY meta_dia DESC
